254 async def _cache_data(self): 

255 """Cache documents and embeddings""" 

256 # Cache embeddings 

257 embeddings_data = { 

258 'embeddings': [doc.embedding for doc in self.documents if doc.embedding is not None], 

259 'titles': [doc.title for doc in self.documents if doc.embedding is not None] 

260 } 

261 with open(self.embeddings_cache, 'wb') as f: 

262 pickle.dump(embeddings_data, f) 

263 

264 # Cache documents 

265 docs_data = [ 

266 { 

267 'title': doc.title, 

268 'content': doc.content, 

269 'url': doc.url, 

270 'category': doc.category 

271 } 

272 for doc in self.documents 

273 ] 

274 with open(self.docs_cache, 'w') as f: 

275 json.dump(docs_data, f, indent=2)

277 async def _load_cached_data(self): 

278 """Load cached documents and embeddings""" 

279 # Load documents 

280 with open(self.docs_cache, 'r') as f: 

281 docs_data = json.load(f) 

282 

283 # Load embeddings 

284 with open(self.embeddings_cache, 'rb') as f: 

285 embeddings_data = pickle.load(f) 

286 

287 # Reconstruct documents with embeddings 

288 for i, doc_data in enumerate(docs_data): 

289 doc = StripeDocument( 

290 title=doc_data['title'], 

291 content=doc_data['content'], 

292 url=doc_data['url'], 

293 category=doc_data['category'] 

294 ) 

295 if i < len(embeddings_data['embeddings']): 

296 doc.embedding = embeddings_data['embeddings'][i] 

297 self.documents.append(doc)

299 async def query(self, question: str, language: str = "python") -> str: 

300 """Query the RAG system with a question""" 

301 if not self.documents: 

302 raise Exception("RAG engine not initialized. Run 'smartpaydoc init' first.") 

303 

304 # Generate query embedding 

305 try: 

306 response = await self.client.embeddings.create( 

307 model="text-embedding-ada-002", 

308 input=question 

309 ) 

310 query_embedding = np.array(response.data[0].embedding) 

311 except Exception as e: 

312 raise Exception(f"Failed to generate query embedding: {e}") 

313 

314 # Find most relevant documents 

315 relevant_docs = self._find_relevant_docs(query_embedding, top_k=3) 

316 

317 # Generate response using LLM 

318 context = "\n\n".join([ 

319 f"**{doc.title}**\n{doc.content}" 

320 for doc in relevant_docs 

321 ]) 

322 

323 prompt = f"""You are SmartPayDoc, an expert Stripe developer assistant. Answer the user's question using the provided Stripe documentation context. 

324 

325Context from Stripe documentation: 

326{context} 

327 

328User question: {question} 

329Preferred language: {language} 

330 

331Guidelines: 

332- Provide accurate, practical answers 

333- Include code examples in the requested language when relevant 

334- Explain concepts clearly 

335- Reference relevant Stripe documentation 

336- If the question can't be answered from the context, say so clearly 

337 

338Answer:"""

340 try: 

341 response = await self.client.chat.completions.create(

347 return response.choices[0].message.content 

348 except Exception as e: 

349 raise Exception(f"Failed to generate response: {e}")

351 def _find_relevant_docs(self, query_embedding: np.ndarray, top_k: int = 3) -> List[StripeDocument]: 

352 """Find the most relevant documents using cosine similarity""" 

353 if not any(doc.embedding is not None for doc in self.documents): 

354 return self.documents[:top_k] # Fallback if no embeddings 

355 

356 similarities = [] 

357 valid_docs = [] 

358 

359 for doc in self.documents: 

360 if doc.embedding is not None: 

361 similarity = cosine_similarity( 

362 query_embedding.reshape(1, -1), 

363 doc.embedding.reshape(1, -1) 

364 )[0][0] 

365 similarities.append(similarity) 

366 valid_docs.append(doc) 

367 

368 # Sort by similarity and return top_k 

369 sorted_indices = np.argsort(similarities)[::-1] 

370 return [valid_docs[i] for i in sorted_indices[:top_k]]
